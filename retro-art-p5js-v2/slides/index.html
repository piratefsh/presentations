grid ‚ø≥

            
  const numRows = 10;
  const numCols = 10;

  // calculate width and height of each cell
  const cellSizeX = width/numRows;
  const cellSizeY = height/numCols;

  // for each cell in grid
  for(let i = 0; i < numRows; i++){
    for(let j = 0; j < numCols; j++){

      // draw ellipse with random size
      const dotSize = random(0, 30);
      ellipse(i*cellSizeX, j*cellSizeY, dotSize, dotSize);
    }
  }

let t = 0; // time

function draw() {
  // set 0,0 to center of canvas
  translate(width/2, height/2);

  // calculate points of a circle
  const x = 150 * sin(t*0.03);  // x = radius * sin(angle);
  const y = 150 * cos(t*0.03);  // y = radius * cos(angle);

  ellipse(x, y, 10, 10);

  // increase time
  t+=1;
}
